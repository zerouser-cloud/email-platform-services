```typescript
@Injectable()
export class PostgresHealthIndicator implements DatabaseHealthIndicator {
  constructor(
    private readonly healthIndicatorService: HealthIndicatorService,
    @Inject(PG_POOL) private readonly pool: Pool,
  ) {}
  async isHealthy(key: string): Promise<HealthIndicatorResult> {
    const indicator = this.healthIndicatorService.check(key);
    try {
      await this.pool.query(PG_HEALTH.QUERY);
      return indicator.up();
    } catch {
      return indicator.down({ message: PG_HEALTH.DOWN_MESSAGE });
    }
  }
}
```

| Boundary | Description |
|----------|-------------|
| Service -> Redis | Service writes/reads cache data via CachePort abstraction |
| CachePort -> ioredis | Internal boundary -- only RedisCacheService touches raw Redis client |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-21-01 | Tampering | cache.service.ts | mitigate | Namespace prefix auto-applied to all keys -- services cannot access other services' keys. No raw client export (D-04). |
| T-21-02 | Information Disclosure | Redis connection | accept | Redis bound to Docker network, not exposed to host in production. REDIS_URL from validated env schema, no hardcoded credentials. |
| T-21-03 | Denial of Service | cache.service.ts | mitigate | Mandatory TTL on every set() prevents unbounded memory growth (D-06). maxRetriesPerRequest limits retry storms. |
| T-21-04 | Tampering | JSON deserialization | mitigate | JSON.parse wrapped in try/catch returning null -- malformed data cannot crash service (research pitfall 3). |
</threat_model>
