2. Add `.refine()` to `GlobalEnvSchema` AFTER the `.object()` call:
```typescript
export const GlobalEnvSchema = z.object({
  ...TopologySchema.shape,
  ...InfrastructureSchema.shape,
  NODE_ENV: z.enum(['development', 'production', 'test']).default('development'),
  PROTO_DIR: z.string().min(1).optional(),
  // ... rest unchanged ...
}).refine(
  (data) => !(data.NODE_ENV === 'production' && data.CORS_ORIGINS === '*'),

    message: 'CORS_ORIGINS cannot be "*" in production. Specify explicit origins (e.g., "https://app.example.com").',
    path: ['CORS_ORIGINS'],
  }
);

export type GlobalEnv = GlobalTopology &

    // ... rest unchanged ...
  };
```
